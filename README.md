#View song details
./mp3_reader -v song.mp3

# Edit the title
./mp3_reader -e -t "New Song" song.mp3

# Edit the artist
./mp3_reader -e -a "Arijit Singh" song.mp3

# Edit the album
./mp3_reader -e -A "Hits 2025" song.mp3

# Edit the year
./mp3_reader -e -y "2025" song.mp3

# Edit the genre
./mp3_reader -e -m "Pop" song.mp3

# Edit the comment
./mp3_reader -e -c "Remastered version" song.mp3

#ask for help
./mp3_reader --help
